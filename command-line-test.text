#  register user
curl --include --request POST --header "Content-Type: application/json" -d '{
  "credentials": {
    "email": "a@c.a",
    "password": "abc123",
    "password_confirmation": "abc123"
  }
}' http://localhost:3000/register


# 2 login user
curl --request POST --header "Content-Type: application/json" -d '{
  "credentials": {
    "email": "user1@example.com",
    "password": "a"
  }
}' http://localhost:3000/login



# 3 logout user
curl --request DELETE --header "Authorization: Token token=fd91384fe88dcb8f34d14f3e2f96961f" http://localhost:3000/logout/1

# 4 list users
curl --header "Authorization: Token token=fd91384fe88dcb8f34d14f3e2f96961f" http://localhost:3000/users

or

curl http://localhost:3000/users


# 5 create a bike
curl --request POST --header "Authorization: Token token=0d4ffe67da0618bb086523543d6c2dcc" --header "Content-Type: application/json" -d '{
  "bike": {
    "title":"The Hold",
    "description":"abc123def456"
  }
}'  http://localhost:3000/bikes

# 5 update a book
curl --request PATCH --header "Authorization: Token token=0d4ffe67da0618bb086523543d6c2dcc" --header "Content-Type: application/json" -d '{
  "bike": {
    "title":"Now Not The Hold",
    "description":"abc123def456"
  }
}'  http://localhost:3000/bikes/16
